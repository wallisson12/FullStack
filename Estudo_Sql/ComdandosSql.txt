create table tbCliente
(
	CPF varchar(11),  Ex: Se eu colocar 'Null', permite que esse campo pode ser nulo
	Nome varchar(100),
    Endereco1 varchar(150),
    Endereco2 varchar(150),
    Bairro varchar(50),
    Cidade varchar(50),
    Estado varchar(50),
    CEP varchar(8),

    Volume_Compra float,
